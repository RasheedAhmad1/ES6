Modules - import and export

Name Export

export const name = 'Rasheed Jan';
import { name } from '...';

Default Export

export default 'value';
import anyName from '...';

Rename Export

export { name as newName };
import { newName } from '...';

Export List + Rename

export {
name1,
name2 as newName2
}
import { name1 as newName1, newName2 } from '...';
